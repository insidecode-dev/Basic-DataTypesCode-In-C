#include <stdio.h>

int main()
{
    
    int a = 4;
    short b = 3;
    char c = 's';
    char c1[]="Hello";
    float flt = 2.3f;
    double d = 56;
    double d1 = 56.45;
    long  l = 72647826;
    long long l1 = 287457825426854;
    long double ld1 = 287362.2874328;
    
    printf("int = %d\n", a);
    printf("short = %hi\n", b);
    printf("char = %c\n", c);
    printf("string = %s\n", c1);
    printf("float = %.1f\n", flt);
    printf("double(int)  = %.lf\n", d);
    printf("double = %.2lf\n", d1);
    printf("long = %ld\n", l);
    printf("long long = %lld\n", l1);
    printf("long double = %.7Lf", ld1);
    
    return 0;
}
